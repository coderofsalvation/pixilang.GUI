
ГЛОБАЛЬНЫЕ ПЕРЕМЕННЫЕ


GUI_TOP - содержит текущий активный объект


ФУНКЦИИ


создать новый объект, все параметры опциональны
GUI_new( $parent, $xpos, $ypos, $width, $height, $color, $transp, $texture )

удалить объект и все под-объекты 
GUI_remove( $obj )

получить позыцию объекта в массиве потомков родителя
GUI_getid( $obj )

создать связь родитель-потомок между двумя объектами
GUI_adopt( $obj, $parent )

перемещает объект в конец очереди отрисовки
GUI_focus( $obj )

побирает стороны текстуры так что-бы она не выглядела сжатой или растяжённой
GUI_fix_ratio( $obj )

распостранить заданную функцию по дереву объектов начиная с входной точки
GUI_propagate( $obj, $function, $flags, $userdata1, $userdata2 )

добавляет строку к объекту
GUI_add_text( $obj, $prop_name, $text, $relxpos, $relypos, $color, $flags )

устанавливает текстуру для объекта
GUI_set_texture( $obj, $texture, $flags, $xoffs, $yoffs, $xcount, $ycount )

обновляет параметры текстуры
GUI_update_texture( $obj )


ОБРАБОТЧИКИ


проверяет условия для взаимодействия с объектом
GUI_check( $obj )

обрабатывает взаимодействия с объектом
GUI_evt( $obj )

обрабатывает прорисовку объекта
GUI_draw( $obj )


СВОЙСТВА (значение по умолчанию в скобках)


свойства объекта, т.е оbj

.x, .y - координаты левого верхнего угла ( 0, 0 )
.cx, .cy - координаты курсора мыши относительно объекта
.w, .h - ширина, длина ( 100, 100 )
.color, .transp - цвет, прозрачность ( get_color( 128, 128, 128), 255 )
.tex - контейнер с текстурой ( -1 )
.parent - содержит текущий объект как потомка ( -1 )
.children - массив под-объектов ( не указано  )
.children.size - размер массива потомков
.str - массив объектов строки ( не указано )
.str.size - размер массива строк
.flags - содержит различные состояния объекта ( 48 ( 16 | 32 ) )
.check - содержыт проверочную функцию ( GUI_check )
.evt - содержит обработчик взаимодействия ( GUI_evt )
.draw - содержит обработчик прорисовки ( GUI_draw )

события которые могут всегда вызыватся

.onevent - вызывается по системному событию
.permanent - вызывается постоянно

события которые могут вызыватся только в обычном режиме

.onpress - вызывается по нажатию мышы внутри объекта
.onrelease - вызывается по нажатию и отпусканию мышы внутри объекта
.onenter - вызывается когда курсор мыши входит в объект
.onleave - вызывается когда курсор мыши выходит за рамки объекта
.onhover - вызывается пока кнопка мыши зажата внутри объекта
.onhold - вызывается пока курсор мыши внутри объекта
.onscrresize - вызывается при изменении размеров окна программы

события которые могут вызыватся только в режиме редактирования
ЛКМ - изменить размер, СКМ - сфокусировать, ПКМ - перетянуть

.onresize - вызывается при изменении размеров объекта посредством мыши
.onfocus - вызывается когда функция GUI_focus применяется к объекту
.ondrag - вызывается при перетягивании объекта посредством мыши

свойства строки, т.e obj.str[ i ]

.text - текст строки ( "" )
.x, .y - коордитаны строки по отношению к объекту ( $obj.w / 2, $obj.h / 2 )
.color - цвет строки ( WHITE )
.flags - содержит различные состояния строки ( 0 )

свойства текстуры, т.e obj.tex

.x, .y - коордитаны по отношению к объекту
.mask - маска цвета ( WHITE )
.xscale, .yscale - горизонтальное и вертикальное масштабирование
.xoffs, .yoffs - смещение внутри текстуры ( 0, 0 )
.xcount, .ycount - пользовательский размер смещения ( get_xsize( obj.tex ), get_ysize( obj.tex ) )
.fxcount, .fycount - размер смещения при прорисовке
.flags - содержит различные состояния текстуры ( 0 )
.effect - еффект который будет применятся к текстуре при прорисовке ( не указано )


ФЛАГИ


флаги объекта

0b0x000000 - режим редактирования: 0 - выкл., 1 - вкл.
0b00x00000 - объект: 0 - спрятать, 1 - показать
0b000x0000 - объект: 0 - блокирован, 1 - активен
0b0000x000 - курсор мыши: 0 - снаружи, 1 - внутри
0b00000x00 - ПКМ\
0b000000x0 - СКМ : 0 - нажата, 1 - отпущена 
0b0000000x - ЛКМ/

флаги строки

0b000x0000 - строка: 0 - спрятать, 1 - показать
0b0000x000 - RIGHT \
0b00000x00 - LEFT   : встроенные флаги
0b000000x0 - BOTTOM : выравнивания текста
0b0000000x - TOP   /

флаги текстуры

0b000000x0 - еффект текстуры: 0 - выкл., 1 - вкл.
0b0000000x - соотношение сторон: 0 - не сохранять, 1 - сохранять

флаги для функции GUI_propagate

0b000000x0 - входная точка: 0 - не пропускать, 1 - пропустить
0b0000000x - распостранение функции: 0 - вниз по дереву, 1 - вверх по дереву
